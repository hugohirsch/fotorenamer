36      /**
37       * Returns the requested tag as String from the image file.
38       *
39       * @param image Image file to extract Metadata from.
40       * @param tag   Tag to extract from the given file, @see TiffConstants
41       * @return Returns exif tag value, in case of any errors the value is an
42       *         empty String.
43       * @throws IOException        if file cannot be accessed.
44       * @throws ImageReadException if an error occurred during image processing.
45       */
46      public static String getExifMetadata(final File image,
47                                           final TagInfo tag)
48              throws IOException, ImageReadException {
49          assert image != null : "Parameter image must not be null";
50          assert tag != null : "Parameter tag must not be null";
51  
52          String result = EMPTY_STRING;
53          IImageMetadata metadata = Sanselan.getMetadata(image);
54          if (metadata instanceof JpegImageMetadata) {
55              JpegImageMetadata jpegMetadata = (JpegImageMetadata) metadata;
56              TiffField field = jpegMetadata.findEXIFValue(tag);
57              if (field != null) {
58                  result = field.getValueDescription();
59                  LOG.info("extraction of " + tag.getDescription()
60                          + " yields " + result);
61              }
62          }
63          return result == null ? EMPTY_STRING : result;
64      }

66      /**
67       * Helper to extract the date this image was created to be used during the
68       * renaming process.
69       * <p/>
70       * In the EXIF standard itself the following convention for dates as is
71       * defined: <i> D. Other Tags DateTime The date and time of image creation.
72       * In this standard it is the date and time the file was changed. The format
73       * is "YYYY:MM:DD HH:MM:SS" with time shown in 24-hour format, and the date
74       * and time separated by one blank character [20.H]. When the date and time
75       * are unknown, all the character spaces except colons (":") may be filled
76       * with blank characters, or else the Interoperability field may be filled
77       * with blank characters. The character string length is 20 bytes including
78       * NULL for termination. When the field is left blank, it is treated as
79       * unknown. Tag = 306 (132.H) Type = ASCII Count = 20 Default = none </i>
80       *
81       * @param image Image to extract metadata from.
82       * @return the date this image was created if found, format is
83       * @throws ParseException     If metadata cannot be parsed correctly.
84       * @throws ImageReadException If image cannot be read.
85       * @throws IOException        If an error occurs when accessing the image's
86       *                            metadata.
87       * @see <a href="http://www.exif.org/samples/canon-ixus.html">Canon EXIF
88       *      example page</a>
89       * @see <a href="http://www.exif.org/specifications.html">EXIF
90       *      specifications</a>
91       * @see <a href="http://www.exif.org/Exif2-2.PDF">EXIF2-2.pdf
92       *      specification</a>
93       */
94  
95      public static String generateCreationDateInCorrectFormat(final File image)
96              throws ImageReadException, IOException, ParseException {
97          String dateValue =
98                  getExifMetadata(image, TiffConstants.EXIF_TAG_CREATE_DATE);
99          if (dateValue != null) {
100             LOG.info("EXIF date value is: " + dateValue);
101 
102             assert dateValue.length() == VALID_EXIF_DATE_LENGTH
103                     : "Invalid length of EXIF metadata, "
104                     + "not complying to the standard";
105 
106             // Date parsing with apache.DateUtils or JDK-DateFormats
107             // does not work due to '-signs in the date string
108             // (unparseable pattern is "'yyyy:MM:dd HH:mm:ss'")
109 
110             // replace special characters to extract digits only
111             dateValue = dateValue.replaceAll(APOSTROPHE, EMPTY_STRING);
112             dateValue = dateValue.replaceAll(COLON, EMPTY_STRING);
113             dateValue = dateValue.replaceAll(SPACE, UNDERSCORE);
114             dateValue += UNDERSCORE;
115             //convert '2011:01:30 13:11:02' to "yyyyMMdd_HHmm_"+fileName
116             dateValue += image.getName();
117 
118             LOG.info("Target filename is: " + dateValue);
119             return dateValue;
120         }
121         LOG.info("No creation date extracted from file " + image);
122 
123         return EMPTY_STRING;
124     }
