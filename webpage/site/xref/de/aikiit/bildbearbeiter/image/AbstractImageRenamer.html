20  public abstract class AbstractImageRenamer implements Runnable {

39      /**
40       * Übernimmt ein Verzeichnis. Dessen Dateien werden in
41       * ZuletztGeändertDatum_ZuletztGeändertUhrzeit_Dateiname umbenannt.<br> Bei
42       * den IXUS-Bildern bedeutet das, dass den Dateinamen der Zeitpunkt der
43       * Bildaufnahme mit hinzugefügt wird. <br>
44       *
45       * @param verzeichnis Directory to work on.
46       * @throws InvalidDirectoryException If there's a problem with the directory
47       *                                   selected.
48       * @throws NoFilesFoundException     if the selected directory is empty.
49       */
50      public AbstractImageRenamer(final String verzeichnis)
51              throws InvalidDirectoryException, NoFilesFoundException {
52          this.currentDirectory = new File(verzeichnis);
53  
54          // erst starten, wenn die Eingabeprüfung erfolgreich war
55          validateUserSelectionAndInitInternally();
56  
57          new Thread(this).start();
58      } // end of Konstruktor

60      /**
61       * Checks whether the selected directory contains any relevant files and
62       * sets internal state appropriately.
63       *
64       * @throws de.aikiit.bildbearbeiter.exception.NoFilesFoundException
65       *          If selected directory contains no files.
66       * @throws de.aikiit.bildbearbeiter.exception.InvalidDirectoryException
67       *          If any I/O-error occured when accessing the directory.
68       */
69      protected final void validateUserSelectionAndInitInternally()
70              throws NoFilesFoundException, InvalidDirectoryException {
71          // Verzeichnis gültig ?
72          if (!this.currentDirectory.isDirectory()) {
73              throw new InvalidDirectoryException(this.currentDirectory);
74          }
75  
76          // Dateien da ?
77          this.imageList = this.currentDirectory.listFiles(
78                  new ImageFilenameFilter());
79          if (this.imageList == null || this.imageList.length == 0) {
80              throw new NoFilesFoundException(this.currentDirectory);
81          }
82  
83          // internen Zustand setzen
84          this.amountOfFiles = this.imageList.length;
85      } // end of pruefeEingabe
86  
87      /**
88       * Performs the actual/technical renaming.
89       *
90       * @throws RenamingErrorException if any errors occur.
91       * @see #validateUserSelectionAndInitInternally()
92       */
93      public final void renameFiles() throws RenamingErrorException {
94          String targetFilename = "";
95          for (int i = 0; i < this.amountOfFiles; i++) {
96  
97              // Daten holen und umbenenen
98              targetFilename = renameImage(this.imageList[i]);
99  
100             // ProgressBar updaten...
101             this.progressBar.setProgress(i);
102             this.progressBar.setText(this.imageList[i].getName());
103 
104             // Da die Namen verschieden lang sind den ProgressBar updaten!
105             LOG.info("Renaming " + this.imageList[i].getName() + " to "
106                     + targetFilename);
107             this.progressBar.updateUI();
108 
109             // only renameFiles files
110             if (this.imageList[i].isFile() && !this.imageList[i].renameTo(
111                             new File(this.imageList[i].getParent(),
112                                     targetFilename))) {
113                 throw new RenamingErrorException("\tFehler bei Bild"
114                         + this.imageList[i].getName());
115             } // end if - isFile()
116         } // end of for
117     } // end of renameFiles

135     public final void run() {
136         String meldung = "";
137         this.progressBar = new ProgressBar(this.amountOfFiles);
138 
139         try {
140             renameFiles();
141         } catch (RenamingErrorException uf) {
142             JOptionPane.showMessageDialog(null,
143                     "Während der Bearbeitung der Datei\n"
144                             + uf.getMessage() + " trat ein Fehler beim "
145                             + "Umbennen auf.",
146                     "Fehler beim Umbenennen",
147                     JOptionPane.ERROR_MESSAGE);
148             return;
149         } // end of catch uf
150 
151         this.progressBar.dispose();
152 
153         // Erfolgsmeldung geben
154         if (this.amountOfFiles == 0) {
155             meldung = "Im Verzeichnis: " + this.currentDirectory.getName()
156                     + "\nwurden keine Dateien\numbenannt.\n\n";
157         } else if (this.amountOfFiles == 1) {
158             meldung = "\nEs wurde eine Datei\n"
159                     + "im Verzeichnis: " + this.currentDirectory.getName()
160                     + "\nerfolgreich umbenannt.\n\n";
161         } else {
162             meldung = "\nEs wurden " + this.amountOfFiles + " Dateien\n"
163                     + "im Verzeichnis: " + this.currentDirectory.getName()
164                     + "\nerfolgreich umbenannt.\n\n";
165         } // end of else
166         JOptionPane.showMessageDialog(null, meldung, "Erfolg",
167                 JOptionPane.INFORMATION_MESSAGE);
168     } // end of run
